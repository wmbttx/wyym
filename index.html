🏗️ 面向对象编程

类和对象示例

class Rectangle {
private:
    int width, height;
public:
    Rectangle(int w, int h) : width(w), height(h) {}
    
    int area() {
        return width * height;
    }
};

int main() {
    Rectangle rect(5, 3);
    cout << "面积: " << rect.area(); // 输出15
    return 0;
}
